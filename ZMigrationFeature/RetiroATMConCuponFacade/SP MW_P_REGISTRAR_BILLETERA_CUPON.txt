PROCEDURE MW_P_REGISTRAR_BILLETERA_CUPON
(
PV_RETRIEVAL_REF IN VARCHAR2,
PV_TERMINAR_REFERENCE IN VARCHAR2,
PV_TERMINAL_ID IN VARCHAR2,
PV_CUPON_ID IN VARCHAR2,
PV_AMOUNT IN VARCHAR2,
PV_CURRENCY IN VARCHAR2,
PV_CODIGO_ERROR OUT VARCHAR2,
PV_MENSAJE_ERROR OUT VARCHAR2
)
AS

BEGIN

INSERT INTO MW_REGISTRO_BILLE_CUPON_ATM
(
RETRIEVAL_REF_NUMBER_37,
TERMINAR_REFERENCE_11,
TERMINAL_ID_41,
CUPON_ID_48,
AMOUNT_04,
CURRENCY_49
)
VALUES
(
LTRIM(RTRIM(PV_RETRIEVAL_REF)),
LTRIM(RTRIM(PV_TERMINAR_REFERENCE)),
LTRIM(RTRIM(PV_TERMINAL_ID)),
LTRIM(RTRIM(PV_CUPON_ID)),
LTRIM(RTRIM(PV_AMOUNT)),
LTRIM(RTRIM(PV_CURRENCY))
);
COMMIT;
PV_CODIGO_ERROR := 'SUCCESS';
PV_MENSAJE_ERROR := '';

EXCEPTION
WHEN OTHERS THEN
PV_CODIGO_ERROR := 'ERROR';
PV_MENSAJE_ERROR := SQLERRM || ' ERROR';
rollback;

END MW_P_REGISTRAR_BILLETERA_CUPON;
