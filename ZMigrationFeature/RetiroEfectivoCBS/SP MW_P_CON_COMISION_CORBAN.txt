CREATE OR REPLACE PROCEDURE MIDDLEWARE.MW_P_CON_COMISION_CORBAN(
    Pv_MONTO                 IN VARCHAR2,
    Pv_MONEDA                IN VARCHAR2,
    Pv_CODIGO_TRANSACCION    IN VARCHAR2,
    Pv_USUARIO_TRANSACCION   IN VARCHAR2,
    Pv_MONTO_COMISION       OUT VARCHAR2,
    Pv_MONEDA_COMISION      OUT VARCHAR2,
    Pv_CODIGO_ERROR         OUT VARCHAR2,
    Pv_MENSAJE_ERROR        OUT VARCHAR2) AS
    --
    Ln_MONTO_COMISION MW_COMISIONES_CORBAN.COMISION%TYPE;
    Lv_MONEDA_COMISION MW_COMISIONES_CORBAN.MONEDA%TYPE;
    --
BEGIN
    --
    SELECT COMISION, MONEDA
      INTO Ln_MONTO_COMISION, Lv_MONEDA_COMISION
      FROM MW_COMISIONES_CORBAN
     WHERE TIPO_TRANSACCION = Pv_CODIGO_TRANSACCION 
       AND MONEDA = Pv_MONEDA
       AND Pv_MONTO BETWEEN MONTO_MINIMO AND MONTO_MAXIMO
       AND USUARIO_TRANSACCION = Pv_USUARIO_TRANSACCION;
    --
    Pv_MONEDA_COMISION := Lv_MONEDA_COMISION;
    Pv_MONTO_COMISION := Ln_MONTO_COMISION;
    Pv_CODIGO_ERROR := 'SUCCESS';
    --   
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        Pv_CODIGO_ERROR:= 'ERROR';
        Pv_MENSAJE_ERROR:= 'No se encontró parametrización de comisión para el tipo de transacción '||Pv_CODIGO_TRANSACCION||' moneda '||Pv_MONEDA||' y monto '||Pv_MONTO ;
        RETURN;
     WHEN OTHERS THEN
        Pv_CODIGO_ERROR:= 'ERROR';
        Pv_MENSAJE_ERROR:= 'Se encontró más de una parametrización de comisión para el tipo de transacción '||Pv_CODIGO_TRANSACCION||' moneda '||Pv_MONEDA||' y monto '||Pv_MONTO ;
        RETURN;
END MW_P_CON_COMISION_CORBAN;
